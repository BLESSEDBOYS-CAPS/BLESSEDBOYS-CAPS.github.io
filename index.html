<div id="contador">Cargando...</div>

<script>

function obtenerCodigoDesdeURL() {
const params = new URLSearchParams(window.location.search);
return params.get('codigo');
}

const codigoGorra = obtenerCodigoDesdeURL();

const SUPABASE_URL = "https://vbgyrfgeyrqcphnuspdb.supabase.co/rest/v1";
const SUPABASE_KEY = "sb_publishable_JFfDRaXIJ2zsre2rPJ_9Vw_1VCM9HEN";

async function actualizarContador() {

if (!codigoGorra) {
document.getElementById("contador").innerText = "Sin código";
return;
}

try {

const respuesta = await fetch(
`${SUPABASE_URL}/CHIPSGORRAS?IDGORRA=eq.${codigoGorra}&select=*`,
{
headers: {
"apikey": SUPABASE_KEY,
"Authorization": `Bearer ${SUPABASE_KEY}`
}
}
);

const datos = await respuesta.json();

if (!datos || datos.length === 0) {
document.getElementById("contador").innerText = "No existe";
return;
}

const imagen = datos[0].IMAGEN;

if (imagen) {
document.body.style.backgroundImage = `url('${imagen}')`;
}

let contadorActual = Number(datos[0].CONTADOR || 0);

contadorActual++;

const actualizar = await fetch(
`${SUPABASE_URL}/CHIPSGORRAS?IDGORRA=eq.${codigoGorra}`,
{
method: "PATCH",
headers: {
"apikey": SUPABASE_KEY,
"Authorization": `Bearer ${SUPABASE_KEY}`,
"Content-Type": "application/json",
"Prefer": "return=representation"
},
body: JSON.stringify({
CONTADOR: contadorActual
})
}
);

if (!actualizar.ok) {
document.getElementById("contador").innerText = "Error UPDATE";
return;
}

document.getElementById("contador").innerText = contadorActual;

} catch (error) {

console.error(error);
document.getElementById("contador").innerText = "Error";

}

}

actualizarContador();

</script>
